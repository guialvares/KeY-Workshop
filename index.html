Storage 


function f() {
	alice.account.balance = 10;
	bob.account.balance = 20;
	Account storage acc = alice.account;

	alice.account = bob.account;
	assert(acc.balance == 20);

	acc.balance = 30;
	assert(alice.account.balance == 30);
	assert(bob.account.balance   == 20);
}
acc <- alice.account
